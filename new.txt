  <script>
    let selectedNumber = null;
    let themeIndex = 0;

    const types = {
      BOY: [
        "Handsome", "Smart", "Brave", "Witty", "Excellent",
        "Charming", "Clever", "Superstar", "Good", "Genius"
      ],
      GIRL: [
        "Beautiful", "Smart", "Creative", "Witty", "Excellent",
        "Kind", "Clever", "Amazing", "Good", "Brilliant"
      ]
    };

    const themes = [
      "linear-gradient(to right, #fceabb, #f8b500)",
      "linear-gradient(to right, #a1c4fd, #c2e9fb)",
      "linear-gradient(to right, #fbc2eb, #a6c1ee)",
      "linear-gradient(to right, #d4fc79, #96e6a1)",
      "linear-gradient(to right, #ffecd2, #fcb69f)"
    ];

    const avatars = {
      BOY: "https://cdn-icons-png.flaticon.com/512/4140/4140048.png",
      GIRL: "https://cdn-icons-png.flaticon.com/512/4140/4140051.png"
    };

    function showStep(step) {
      document.querySelectorAll('.container > div').forEach(div => div.classList.add('hidden'));
      document.getElementById(step).classList.remove('hidden');

      if (step === 'character') {
        document.getElementById('avatar').src = avatars.BOY;
      }
    }

    function selectNumber(num) {
      selectedNumber = num;
      showStep('character');
    }

    function selectCharacter(gender) {
      const type = types[gender][selectedNumber - 1];
      document.getElementById('userType').textContent = `You are ${type}!`;
      document.getElementById('avatar').src = avatars[gender];
      showStep('result');
      launchConfetti();
    }

    function restart() {
      selectedNumber = null;
      themeIndex = (themeIndex + 1) % themes.length;
      document.body.style.background = themes[themeIndex];
      showStep('welcome');
    }

    function playSound() {
      document.getElementById('clickSound').play();
    }

    function launchConfetti() {
      confetti({
        particleCount: 150,
        spread: 70,
        origin: { y: 0.6 }
      });
    }
